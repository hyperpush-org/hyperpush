fn handle_text(request) do

end

fn handle_json(request) do
  HTTP.response(200, "{\"message\":\"Hello, World!\"}")
end

fn main() do
  HTTP.serve((HTTP.router()
    |> HTTP.on_get("/text", handle_text)
    |> HTTP.on_get("/json", handle_json)), 3000)
end

Create `benchmarks/go/main.go` — an idiomatic Go HTTP server using stdlib `net/http` (no framework, per user decision).

The Go server must:
- Listen on port 3001
- Handle GET /text: respond 200 with Content-Type: text/plain and body "Hello, World!\n"
- Handle GET /json: respond 200 with Content-Type: application/json and body {"message":"Hello, World!"}
- Use GOMAXPROCS set to runtime.NumCPU() for fair CPU utilization
- Use http.DefaultServeMux (or a local mux)

```go
package main

import (
	"fmt"
	"net/http"
	"runtime"
)

func textHandler(w http.ResponseWriter, r *http.Request) {
	w.Header().Set("Content-Type", "text/plain")
	fmt.Fprint(w, "Hello, World!\n")
}

func jsonHandler(w http.ResponseWriter, r *http.Request) {
	w.Header().Set("Content-Type", "application/json")
	fmt.Fprint(w, `{"message":"Hello, World!"}`)
}

func main() {
	runtime.GOMAXPROCS(runtime.NumCPU())
	mux := http.NewServeMux()
	mux.HandleFunc("/text", textHandler)
	mux.HandleFunc("/json", jsonHandler)
	http.ListenAndServe(":3001", mux)
}
```

Also create `benchmarks/go/go.mod` with a minimal module declaration so the runner script can use `go run .`:
```

```

No external dependencies; stdlib-only. The go.mod enables `go run .` invocation from the benchmarks/go/ directory.
